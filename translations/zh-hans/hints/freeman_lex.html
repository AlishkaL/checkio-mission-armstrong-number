def is_armstrong(num: int) -> bool:

    text = str(num)
    l = len(text)
    res = 0
    for digit in text:
        res += int(digit) ** l
        
    return res == num

在不转换为字符串的情况下，通过连续除以 10 创建一个数字列表。 

def is_armstrong(num: int) -> bool:

    temp, digits = num, []
    while temp:
        temp, digit = divmod(temp, 10)
        digits.append(digit)
    l = len(digits)

    return sum(d**l for d in digits) == num

使用 walrus 运算符在一行中声明两个变量，并使用 map 函数对每个数字应用某些函数。 

def is_armstrong(num: int) -> bool:

    l = len(text:= str(num))
    
    return sum(map(lambda i: int(i)**l, text)) == num
